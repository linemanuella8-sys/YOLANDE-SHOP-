rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users: only the authenticated user can read/write their own document
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Carts: user-specific carts stored under users or top-level carts collection
    match /carts/{cartId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId || request.auth.uid == request.resource.data.userId;
    }

    // Orders: authenticated users can create and read their own orders; admins only can manage all
    match /orders/{orderId} {
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      allow update, delete: if false; // managed by backend/cloud functions
    }

    // Products: public read, admin-only write (admin determined by custom claim 'admin' == true)
    match /products/{productId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.admin == true;
    }

    // Messages: user can create messages (contact/support), read own messages
    match /messages/{msgId} {
      allow create: if request.auth != null;
      allow read: if request.auth != null && (resource.data.userId == request.auth.uid || request.auth.token.admin == true);
      allow update, delete: if request.auth != null && request.auth.token.admin == true;
    }

    // Notifications: user notifications under users/{uid}/notifications or top-level notifications
    match /notifications/{notifId} {
      allow read: if request.auth != null && (resource.data.userId == request.auth.uid || request.auth.token.admin == true);
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
      allow update: if request.auth != null && request.auth.token.admin == true;
    }

    // Loyalty & favorites stored under user doc — enforce same rule as users
    match /{document=**} {
      // Fallback: deny by default
      allow read, write: if false;
    }
  }
}
